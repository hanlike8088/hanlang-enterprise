  // ---- Complaints ----
  async createComplaint(dto: CreateComplaintDto) {
    const count = await this.prisma.crmComplaint.count();
    const code = `KS-${new Date().getFullYear()}-${String(count + 1).padStart(4, '0')}`;
    await this.getCustomer(dto.customerId);
    if (dto.orderId) { await this.getOrder(dto.orderId); }
    return this.prisma.crmComplaint.create({
      data: { complaintCode: code, ...dto },
      include: { customer: { select: { id: true, customerName: true } }, order: { select: { id: true, orderCode: true } } },
    });
  }

    if (severity) where.severity = severity;

    return this.prisma.crmComplaint.findMany({

  // ---- Reconciliations ----
  async createReconciliation(dto: CreateReconciliationDto) {
    const count = await this.prisma.crmReconciliation.count();
    const code = `DZ-${new Date().getFullYear()}-${String(count + 1).padStart(4, '0')}`;
    await this.getCustomer(dto.customerId);
    if (dto.orderId) await this.getOrder(dto.orderId);
    const paid = dto.paidAmount || 0;

    return this.prisma.crmReconciliation.create({
      data: { reconciliationCode: code, ...dto, paidAmount: paid, balance },
      include: { customer: { select: { id: true, customerName: true } }, order: { select: { id: true, orderCode: true } } },
    });
  }

  async updateReconciliation(id: string, dto: UpdateReconciliationDto) {
    const existing = await this.getReconciliation(id);
    const data: any = {};
    if (dto.totalAmount !== undefined) data.totalAmount = dto.totalAmount;
    if (dto.paidAmount !== undefined) data.paidAmount = dto.paidAmount;
    if (dto.paymentDueDate !== undefined) data.paymentDueDate = new Date(dto.paymentDueDate);
    if (dto.status !== undefined) data.status = dto.status;
    if (dto.notes !== undefined) data.notes = dto.notes;
    if (data.totalAmount !== undefined || data.paidAmount !== undefined) {
      data.balance = (data.totalAmount ?? existing.totalAmount) - (data.paidAmount ?? existing.paidAmount);
    }
    return this.prisma.crmReconciliation.update({
      where: { id }, data,
      include: { customer: { select: { id: true, customerName: true } }, order: { select: { id: true, orderCode: true } } },
    });
  }

  // ---- Payments ----
  async createPayment(dto: CreatePaymentDto) {
    const count = await this.prisma.crmPayment.count();
    const code = `HK-${new Date().getFullYear()}-${String(count + 1).padStart(4, '0')}`;
    await this.getCustomer(dto.customerId);
    if (dto.reconciliationId) await this.getReconciliation(dto.reconciliationId);
    const payment = await this.prisma.crmPayment.create({
      data: { paymentCode: code, ...dto, paymentDate: dto.paymentDate ? new Date(dto.paymentDate) : new Date() },
      include: { customer: { select: { id: true, customerName: true } }, reconciliation: { select: { id: true, reconciliationCode: true } } },
    });
    if (dto.reconciliationId) {
      const rec = await this.prisma.crmReconciliation.findUnique({ where: { id: dto.reconciliationId } });
      if (rec) {
        const paid = rec.paidAmount + dto.amount;
        const balance = rec.totalAmount - paid;
        await this.prisma.crmReconciliation.update({ where: { id: dto.reconciliationId }, data: { paidAmount: paid, balance, status: balance <= 0 ? 'paid' : 'partial' } });
      }
    }
    return payment;
  }

  async getPayments(keyword?: string, customerId?: string, reconciliationId?: string) {
    const where: any = {};
    if (keyword) { where.OR = [{ paymentCode: { contains: keyword } }, { referenceNo: { contains: keyword } }]; }
    if (customerId) where.customerId = customerId;
    if (reconciliationId) where.reconciliationId = reconciliationId;
    return this.prisma.crmPayment.findMany({
      where, include: { customer: { select: { id: true, customerName: true } }, reconciliation: { select: { id: true, reconciliationCode: true } } },
      orderBy: { paymentDate: 'desc' },
    });
  }

  async getPayment(id: string) {
    const p = await this.prisma.crmPayment.findUnique({
      where: { id }, include: { customer: { select: { id: true, customerName: true } }, reconciliation: { select: { id: true, reconciliationCode: true } } },
    });
    if (!p) throw new NotFoundException('回款记录不存在');
    return p;
  }

  async updatePayment(id: string, dto: UpdatePaymentDto) {
    const existing = await this.getPayment(id);
    const payment = await this.prisma.crmPayment.update({
      where: { id }, data: { ...dto, paymentDate: dto.paymentDate ? new Date(dto.paymentDate) : undefined },
    });
    if (existing.reconciliationId) {
      const payments = await this.prisma.crmPayment.findMany({ where: { reconciliationId: existing.reconciliationId } });
      const totalPaid = payments.reduce((sum: number, p: any) => sum + p.amount, 0);
      const rec = await this.prisma.crmReconciliation.findUnique({ where: { id: existing.reconciliationId } });
      if (rec) {
        const balance = rec.totalAmount - totalPaid;
        await this.prisma.crmReconciliation.update({ where: { id: existing.reconciliationId }, data: { paidAmount: totalPaid, balance, status: balance <= 0 ? 'paid' : 'partial' } });
      }
    }
    return payment;
  }

  async deletePayment(id: string) {
    const existing = await this.getPayment(id);
    const result = await this.prisma.crmPayment.delete({ where: { id } });
    if (existing.reconciliationId) {
      const payments = await this.prisma.crmPayment.findMany({ where: { reconciliationId: existing.reconciliationId } });
      const totalPaid = payments.reduce((sum: number, p: any) => sum + p.amount, 0);
      const rec = await this.prisma.crmReconciliation.findUnique({ where: { id: existing.reconciliationId } });
      if (rec) {
        const balance = rec.totalAmount - totalPaid;
        await this.prisma.crmReconciliation.update({ where: { id: existing.reconciliationId }, data: { paidAmount: totalPaid, balance, status: totalPaid <= 0 ? 'pending' : balance <= 0 ? 'paid' : 'partial' } });
      }
    }
    return result;
  }

  async getPaymentsForCustomer(customerId: string) {
    return this.prisma.crmPayment.findMany({ where: { customerId }, orderBy: { paymentDate: 'desc' } });
  }
